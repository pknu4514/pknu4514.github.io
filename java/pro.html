package org.javaro.project;

import org.javaro.project.Customer;

public class Day {

private String time;

private String pet;

private Customer customer;

public Day(String time) { this.time=time;}

public void setTime(String time) {this.time = time;}

public String getTime() {return this.time;}

public void setPet(String pet) {this.pet = pet;}

public String getPet() {return this.pet;}

public void setCustomer(Customer customer) {this.customer= customer;}

public Customer getCustomer() {return this.customer;}

public String toString() {

String available;

if(this.getCustomer() == null) {

available = "예약가능";

}else {

available ="견주=" +this.getCustomer().getName();

}

return "예약날짜="+this.getTime()+",강아지 이름="+this.getPet()+","+available;

}

}

MyShop 클래스 코드

package org.javaro.project;

import java.util.ArrayList;

import org.javaro.project.Day;

import org.javaro.project.Customer;

public class MyShop {

public String storeName;

public ArrayListdays;

public ArrayListcustomers;

public MyShop(String storeName) {

this.storeName = storeName;

days = new ArrayList();

customers=new ArrayList();

}

public String getStoreName() {

return this.storeName;}

public ArrayList getDays(){

return this.days;}

public ArrayListgetCustomers(){

return this.customers;}

public void addDay(Day day) {

this.getDays().add(day);}

public void removeDay(Day day) {

this.getDays().remove(day);

}

public void addCustomer(Customer customer) {

this.customers.add(customer);

}

public void removeCustomer(Customer customer) {

this.customers.remove(customer);

}

public boolean checkOut(Day day, Customer customer) {

if(day.getCustomer()==null)

{

day.setCustomer(customer);

return true;

}else {

return false;

}

}

public boolean checkln(Day day) {

if(day.getCustomer() != null) {

day.setCustomer(null);

return true;

}else {

return false;

}

}

public ArrayListgetDaysForCustomer(Customer customer){

ArrayListresult = new ArrayList();

for(Day aDay : this.getDays()) {

if((aDay.getCustomer()!=null)&&(aDay.getCustomer().getName().contentEquals(customer.getName()))) {

result.add(aDay);

}

}

return result;

}

public ArrayListgetAvailableDays(){

ArrayList result = new ArrayList();

for(Day aDay : this.getDays()) {

if(aDay.getCustomer() == null) {

result.add(aDay);

}

}

return result;

}

public ArrayListgetUnavailableDays(){

ArrayListresult = new ArrayList();

for(Day aDay : this.getDays()) {

if(aDay.getCustomer()!=null) {

result.add(aDay);

}

}

return result;

}

public String toString() {

return this.getStoreName()+"의 예약 가능 시간="+

this.getStoreName()+"타임, 회원수="+

this.getCustomers().size()+"명";

}

public void printStatus() {

System.out.println("--애견 미용샵 예약 현황--\n"+this.toString());

for(Day aDay : this.getDays()) {

System.out.println(aDay);

}

for(Customer customer : this.getCustomers()) {

int count = this.getDaysForCustomer(customer).size();

System.out.println(customer+"은/는"+count+"번 예약잡음");

}

System.out.println("현재 예약 가능한 시간:" + this.getAvailableDays().size());

System.out.println("--리포트 종료--");

}

}

MyStore 클래스 코드

package org.javaro.project;

import org.javaro.project.Day;

import org.javaro.project.MyShop;

import org.javaro.project.Customer;

public class MyStore {

public static void main(String[] args) {

MyShop myStore = new MyShop("미용샵");

Day day1 = new Day("part1: 14:00~15:00");

Day day2 = new Day("part2: 16:00~17:00");

day1.setPet("콩콩이");

day2.setPet("초코");

Customer customer1 = new Customer();

Customer customer2 = new Customer();

customer1.setName("이몽룡");

customer2.setName("변학도");

myStore.addDay(day1);

myStore.addDay(day2);

myStore.addCustomer(customer1);

myStore.addCustomer(customer2);

System.out.println("day1날짜를 name2 변학도에게 예약");

myStore.checkOut(day1,customer2);

myStore.printStatus();

System.out.println("오늘자 part1 관리 완료 ");

myStore.checkln(day1);

System.out.println("day2 날짜를 name1 이몽룡에게 예약");

myStore.checkOut(day2, customer1);

myStore.printStatus();

}

}
